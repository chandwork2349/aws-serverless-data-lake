
-- Total sales by month
SELECT year, month, SUM(amount) AS total_sales
FROM curated_sales
GROUP BY year, month
ORDER BY year, month;

-- Region-wise performance
SELECT region, COUNT(*) AS total_orders, SUM(amount) AS revenue
FROM curated_sales
GROUP BY region;

-- Top products
SELECT product, SUM(amount) AS revenue
FROM curated_sales
GROUP BY product
ORDER BY revenue DESC;

-- Average order value
SELECT AVG(amount) AS avg_order_value
FROM curated_sales;

-- Monthly sales trend
SELECT year, month,
SUM(amount) AS revenue
FROM curated_sales
GROUP BY year, month
ORDER BY year, month;
